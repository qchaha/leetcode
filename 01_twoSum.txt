package main

import (
	"fmt"
)

func main() {
	a := []int{1, 2, 3, 4}
	fmt.Println(twoSum(a, 6))
}

func twoSum(nums []int, target int) []int {
	var re []int
	c := 0
	for _,i := range nums {
		c += 1
		tmp := nums[c:4]
		for _,j := range tmp {
		
			if i+j == target {
				re = append(re,i)
				re = append(re,j)
			}
		}
	}
	return re
}
